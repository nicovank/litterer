# SPEC CPU 2000

```bash
sudo mount -t iso9660 -o ro,exec,loop spec2000v1.3.iso /mnt/spec2000
cd /mnt/spec2000
./install.sh
```

## 197.parser

```bash
cp -r ./cpu2000/benchspec/CINT2000/197.parser .
cp -r ./cpu2000/benchspec/Makefile.defaults ./197.parser/
cd ./197.parser/src
cp ../data/ref/input/ref.in .
cp ../data/all/input/2.1.dict .
cp -r ../data/all/input/words .
patch -i ../../197.parser.patch

make clean NOINC=1
make -j NOINC=1 CC=clang-21 CFLAGS='-DSPEC_CPU2000 -g -O3 -march=native -mtune=native -DUSE_MALLOC' LDFLAGS='-flto'
mv parser parser-malloc

make clean NOINC=1
make -j NOINC=1 CC=clang-21 CFLAGS='-DSPEC_CPU2000 -g -O3 -march=native -mtune=native' LDFLAGS='-flto'
mv parser parser-arena

cd ../..
```

Malloc/free:

```bash
# Blank heap:
time ./parser 2.1.dict -batch <ref.in >/dev/null

# Detect:
LD_PRELOAD=../../build/libdetector_distribution.so ./parser 2.1.dict -batch <ref.in >/dev/null

# With littering:
env LD_PRELOAD=../../build/liblitterer_distribution_standalone.so LITTER_MULTIPLIER=1 LITTER_OCCUPANCY=0.4 ./parser 2.1.dict -batch <ref.in >/dev/null
```


## boxed-sim

```bash
wget -qO- https://pages.cs.wisc.edu/~schenney/research/simulation/boxed-sim/boxed-sim.tar.gz | tar xz --one-top-level=boxed-sim
cd boxed-sim
patch -i ../boxed-sim.patch

make clean
make -j CC=clang-21 CFLAGS='-std=gnu89 -I. -g -O3 -march=native -mtune=native -DUSE_MALLOC' LDFLAGS='-flto -lm'
mv boxed boxed-malloc


make clean
make -j CC=clang-21 CFLAGS='-std=gnu89 -I. -g -O3 -march=native -mtune=native' LDFLAGS='-flto -lm'
```
